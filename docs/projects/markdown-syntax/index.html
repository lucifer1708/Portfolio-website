16func main() {
17	apiurl, err := url.Parse("https://wallhaven.cc/api/v1/search")
18	checkNil(err)
19	values := apiurl.Query()
20	fmt.Println("Fetching Data........")
21	fmt.Println("Downloading will start soon........")
22	str, n := takeArgs()
23	values.Add("q", str)
24	// For loop to increment page number
25	for i := 1; i <= n; i++ {
26		num := strconv.Itoa(i)
27		fmt.Println("Page number: ", num)
28		values.Set("page", num)
29		apiurl.RawQuery = values.Encode()
30		fmt.Println(apiurl.String())
31		// make Get rqst
32		response, err := http.Get(apiurl.String())
33		checkNil(err)
34		defer response.Body.Close()
35		// Read the data from Get rqst
36		data, err := ioutil.ReadAll(response.Body)
37		checkNil(err)
38		var res Response
39		json.Unmarshal(data, &res)
40		// for loop to print all the data from json that is extracted
41		for _, p := range res.Data {
42			_, title := path.Split(p.Path)
43			Downloadfile(title, p.Path)
44			fmt.Println("Downloaded: ", p.Path)
45		}
46
47	}
48}

50// This function takes arguments from the flags specified by user
51func takeArgs() (string, int) {
52	var s string
53	flag.StringVar(&s, "s", "", "Search for given query")
54	page := flag.Int("n", 5, "# of Pages")
55	flag.Parse()
56	n := *page
57	if len(s) != 0 {
58		return s, n
59	} else {
60		fmt.Println("add some argument")
61		os.Exit(1)
62	}
63	return "", n
64}

66// Function to Download file from url. It takes param title i.e. file name , and url.
67func Downloadfile(title string, url string) {
68	// Get the data
69	resp, err := http.Get(url)
70	checkNil(err)
71	defer resp.Body.Close()
72	out, err := os.Create(title)
73	checkNil(err)
74	defer out.Close()
75	_, err = io.Copy(out, resp.Body)
76	if err != nil {
77		fmt.Println(err)
78	}
79}

81// CheckNil function follows the principle of DRY
82func checkNil(err error) {
83	if err != nil {
84		panic(err)
85	}
86}
